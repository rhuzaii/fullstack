<!DOCTYPE html>
<html xmlns="http://www.w3.org/1999/xhtml">
    <head><title>Form Submission</title>
        <script type="text/javascript">
            //<![CDATA[
            function validate() {
                var age=document.getElementById("age").value;
                var weight=document.getElementById("weight").value;

                if(age<=17){
                    alert("Age must be greater than 17");
                    return false;
                }

                if(weight<45 || weight>75){
                    alert("Weight must be between 45 and 75");
                    return false;
                }
                return true;
            }//]]>
        </script>
    </head>
    <body>
        <form onsubmit="return validate();">
            First Name: <input type="text" /><br />
            Last Name: <input type="text" /><br />
            Middle Initial: <input type="text" /><br />
            Age: <input type="number" id="age"/><br />
            Weight: <input type="number" id="weight"/><br />
            <input type="submit" value="Submit" />
        </form>
    </body>
</html>
