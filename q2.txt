<script>
function monthConverter() {
  const months = [
    "January", "February", "March", "April",
    "May", "June", "July", "August",
    "September", "October", "November", "December"
  ];

  // Inner function (closure)
  return function(num) {
    if (isNaN(num)) {
      return "Bad Number";
    }

    num = Math.floor(num); // strip decimal part

    if (num < 1 || num > 12) {
      return "Bad Number";
    }

    return months[num - 1];
  };
}

const getMonthName = monthConverter();

function showMonth() {
  const input = document.getElementById("monthNum").value;
  const result = getMonthName(input);
  document.getElementById("monthName").value = result;
}
</script>
